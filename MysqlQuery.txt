

SELECT * FROM histransz2.digitalsignature // select table
SHOW CREATE PROCEDURE histransz2.pr_savedigitalSignature; // Find table name by store procedure



SELECT 
   SPECIFIC_SCHEMA,SPECIFIC_NAME,PARAMETER_NAME,DATA_TYPE,DTD_IDENTIFIER
   FROM 
   INFORMATION_SCHEMA.PARAMETERS
   WHERE 
   SPECIFIC_SCHEMA = 'hiscdr'
   AND ROUTINE_TYPE = 'PROCEDURE'
   
   
   
   SELECT DISTINCT ROUTINE_NAME, TABLE_NAME
   FROM infomation_schema.ROUTINES
   INNER JOIN infomation_schema.TABLES ON TABLE_SCHEMA = ROUTINE_SCHEMA
   WHERE ROUTINE_TYPE = 'PROCEDURE'
   AND ROUTINE_NAME = '[stored_procedure_name]'
