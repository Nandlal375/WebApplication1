CREATE DEFINER=`root`@`localhost` PROCEDURE `InsertAddStateUser`(IN name VARCHAR(255), IN stateValue VARCHAR(250), IN password VARCHAR(200))
BEGIN
IF NOT exists(SELECT 1 FROM tbl_user WHERE user_name = name) THEN
INSERT INTO tbl_user(user_name, state,user_password,alternate_no) VALUES (name, stateValue,password, '');
ELSE
SELECT 000;
END IF;
END
